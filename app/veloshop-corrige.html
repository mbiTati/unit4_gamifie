🔷 Étape 2 — MCD (Cardinalités)

CATEGORIE ──(1,N)── contient ──(1,1)── VELO
CLIENT    ──(0,N)── passe    ──(1,1)── VENTE
CLIENT    ──(0,N)── demande  ──(1,1)── REPARATION
VELO      ──(0,N)── subit    ──(1,1)── REPARATION
VENTE     ──(1,N)── contient ──(1,1)── LIGNE_VENTE ──(1,1)── concerne ──(0,N)── VELO
LIGNE_VENTE est la table de liaison de la relation N:M entre VENTE et VELO. Elle porte l'attribut "quantité" et "prix_unitaire".

💻 Étape 4 — SQL CREATE TABLE (solution complète)

CREATE TABLE categorie (
  id_categorie INTEGER PRIMARY KEY AUTOINCREMENT,
  nom VARCHAR(50) NOT NULL UNIQUE,
  description TEXT
);

CREATE TABLE client (
  id_client INTEGER PRIMARY KEY AUTOINCREMENT,
  nom VARCHAR(100) NOT NULL,
  prenom VARCHAR(100) NOT NULL,
  telephone VARCHAR(20),
  email VARCHAR(150) NOT NULL UNIQUE,
  adresse VARCHAR(200),
  npa VARCHAR(10),
  ville VARCHAR(100)
);

CREATE TABLE velo (
  id_velo INTEGER PRIMARY KEY AUTOINCREMENT,
  id_categorie INTEGER NOT NULL,
  marque VARCHAR(100) NOT NULL,
  modele VARCHAR(100) NOT NULL,
  type_velo VARCHAR(20) CHECK(type_velo IN ('neuf','occasion')),
  taille VARCHAR(5),
  prix DECIMAL(10,2) NOT NULL CHECK(prix >= 0),
  stock INTEGER NOT NULL DEFAULT 0 CHECK(stock >= 0),
  FOREIGN KEY (id_categorie) REFERENCES categorie(id_categorie)
);

CREATE TABLE reparation (
  id_reparation INTEGER PRIMARY KEY AUTOINCREMENT,
  id_client INTEGER NOT NULL,
  id_velo INTEGER NOT NULL,
  date_depot DATE NOT NULL,
  probleme TEXT NOT NULL,
  statut VARCHAR(20) NOT NULL DEFAULT 'en_cours'
    CHECK(statut IN ('en_cours','terminee','annulee')),
  cout_estime DECIMAL(10,2),
  cout_final DECIMAL(10,2),
  FOREIGN KEY (id_client) REFERENCES client(id_client),
  FOREIGN KEY (id_velo) REFERENCES velo(id_velo)
);

CREATE TABLE vente (
  id_vente INTEGER PRIMARY KEY AUTOINCREMENT,
  id_client INTEGER NOT NULL,
  date_vente DATE NOT NULL,
  remise_pct DECIMAL(5,2) DEFAULT 0 CHECK(remise_pct BETWEEN 0 AND 50),
  total DECIMAL(10,2) NOT NULL,
  FOREIGN KEY (id_client) REFERENCES client(id_client)
);

CREATE TABLE ligne_vente (
  id_vente INTEGER NOT NULL,
  id_velo INTEGER NOT NULL,
  quantite INTEGER NOT NULL DEFAULT 1 CHECK(quantite >= 1),
  prix_unitaire DECIMAL(10,2) NOT NULL,
  PRIMARY KEY (id_vente, id_velo),
  FOREIGN KEY (id_vente) REFERENCES vente(id_vente) ON DELETE CASCADE,
  FOREIGN KEY (id_velo) REFERENCES velo(id_velo)
);
Note : l'éditeur DataQuest utilise SQLite. AUTO_INCREMENT s'écrit AUTOINCREMENT. En MySQL réel : INT AUTO_INCREMENT PRIMARY KEY.

10 Requêtes SELECT

-- 1. Vélos électriques
SELECT * FROM velo WHERE id_categorie = 4;

-- 2. Les 5 vélos les plus chers
SELECT marque, modele, prix FROM velo ORDER BY prix DESC LIMIT 5;

-- 3. Nombre de vélos par catégorie
SELECT c.nom, COUNT(v.id_velo) as nb FROM categorie c
LEFT JOIN velo v ON c.id_categorie = v.id_categorie GROUP BY c.nom;

-- 4. Chiffre d'affaires total
SELECT SUM(total) as ca_total FROM vente;

-- 5. Ventes avec nom client (JOIN 2 tables)
SELECT v.date_vente, cl.nom, cl.prenom, v.total
FROM vente v JOIN client cl ON v.id_client = cl.id_client;

-- 6. Détail ventes (JOIN 3 tables)
SELECT cl.nom, ve.marque, ve.modele, lv.quantite, lv.prix_unitaire
FROM ligne_vente lv
JOIN vente v ON lv.id_vente = v.id_vente
JOIN client cl ON v.id_client = cl.id_client
JOIN velo ve ON lv.id_velo = ve.id_velo;

-- 7. Clients ayant dépensé plus de 2000 CHF
SELECT cl.nom, cl.prenom, SUM(v.total) as total_achats
FROM client cl JOIN vente v ON cl.id_client = v.id_client
GROUP BY cl.id_client HAVING total_achats > 2000;

-- 8. Vélos jamais vendus
SELECT marque, modele FROM velo
WHERE id_velo NOT IN (SELECT id_velo FROM ligne_vente);

-- 9. UPDATE — augmenter prix électriques de 5%
UPDATE velo SET prix = prix * 1.05 WHERE id_categorie = 4;

-- 10. VIEW — stock par catégorie
CREATE VIEW stock_par_categorie AS
SELECT c.nom, COUNT(v.id_velo) as nb_modeles, SUM(v.stock) as stock_total
FROM categorie c LEFT JOIN velo v ON c.id_categorie = v.id_categorie
GROUP BY c.nom;

SELECT * FROM stock_par_categorie;
