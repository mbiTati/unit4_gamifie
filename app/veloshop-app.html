Requête SQL
SELECT c.nom AS categorie,
    COUNT(v.id_velo) AS nb_modeles,
    SUM(v.stock) AS stock_total,
    AVG(v.prix) AS prix_moyen
FROM categorie c
JOIN velo v ON c.id_categorie = v.id_categorie
GROUP BY c.nom
ORDER BY stock_total DESC;
